# DuktapeOrm
Simple TypeScript ORM for Duktape JS engine.

```ts
import { Entity, Column, EntityManager, PrimaryAutoGeneratedKey } from "./DukOrm"

@Entity("UserTable")
class User {
    @PrimaryAutoGeneratedKey()
    id: number;

    @Column()
    login: string;

    @Column()
    password: string;

    @Column({name: "firstName"})
    name: Date;
}

const manager = new EntityManager(User);
let all = manager.select();
let allLimited = manager.select(e => e.login = 'someLogin');
let some = manager.selectSome(['login', 'password']);
let isUserExist = manager.isExist({name: 'Alex'});
```
